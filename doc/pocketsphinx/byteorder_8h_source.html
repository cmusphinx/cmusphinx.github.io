64 /* Macro to byteswap an int16 variable. x = ptr to variable */
65 #define SWAP_INT16(x) *(x) = ((0x00ff & (*(x))>>8) | (0xff00 & (*(x))<<8))

67 /* Macro to byteswap an int32 variable. x = ptr to variable */
68 #define SWAP_INT32(x) *(x) = ((0x000000ff & (*(x))>>24) | \
69  (0x0000ff00 & (*(x))>>8) | \
70  (0x00ff0000 & (*(x))<<8) | \
71  (0xff000000 & (*(x))<<24))
